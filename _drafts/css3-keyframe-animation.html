@keyframes point-fade { 
  from { opacity:0; } 
  to { opacity:1;} 
}
animation:point-fade linear 1; /* call our keyframe named fadeIn, use animattion ease-in and repeat it only 1 time */
animation:point-fade linear 1;
animation-fill-mode: forward; /* this makes sure that after animation is done we remain at the last keyframe value (opacity: 1)*/
